#!/usr/bin/env python3
import sys

for line in sys.stdin:
    line = line.strip()
    year, temp_max, temp_min = map(int, line.split(","))
    print(f"{year}\t{temp_max}\t{temp_min}")


# Create reducer code
nano wered.py

#!/usr/bin/env python3
import sys

dmax = {}
dmin = {}

for line in sys.stdin:
    year, tmax, tmin = map(int, line.strip().split("\t"))

    if year not in dmax:
        dmax[year] = tmax
        dmin[year] = tmin
    else:
        dmax[year] = max(dmax[year], tmax)
        dmin[year] = min(dmin[year], tmin)

for year in dmax:
    print(f"Year: {year}, Max Temp: {dmax[year]}, Min Temp: {dmin[year]}")


# Give execute permission
